10. Основы работы с переменными, условными операторами, циклами, массивами и функциями в JS.

Объявление переменных и условные операторы
const name = prompt("Введите ваше имя:"); const age = Number(prompt("Введите ваш возраст:")); if (age >= 18) { console.log(`${name}, вы совершеннолетний.`); } else { console.log(`${name}, вы несовершеннолетний.`); }
Циклы
console.log("Числа от 1 до 10:"); for (let i = 1; i <= 10; i++) { console.log(i); } console.log("Числа от 10 до 1:"); let num = 10; while (num >= 1) { console.log(num); num--; }
Работа с массивами
const lectures = ['Основы HTML', 'Формы и кнопки', 'HTML таблицы', 'Основы CSS', 'Работа с классами CSS', 'Работа с Flex']; const practices = ['Практика 1', 'Практика 2', 'Практика 3', 'Практика 4', 'Практика 5', 'Практика 6']; lectures.push(prompt("Введите дополнительную тему лекции:")); practices.unshift(prompt("Введите дополнительную тему практики:")); console.log("Темы лекций:"); lectures.forEach((lecture) => console.log(lecture)); console.log("Темы практик:"); practices.forEach((practice) => console.log(practice)); function arrayToString(array) { return array.join(", "); } console.log("Темы лекций (строка):", arrayToString(lectures)); console.log("Темы практик (строка):", arrayToString(practices));
Манипуляции с массивами
const lectureTopics = ['Основы HTML', 'Формы и кнопки', 'HTML таблицы', 'Основы CSS', 'Работа с классами CSS', 'Работа с Flex']; function filterLecturesByLetterO(array) { return array.filter((item) => item.startsWith("О")); } const filteredTopics = filterLecturesByLetterO(lectureTopics); console.log("Темы, начинающиеся с буквы 'О':", filteredTopics);
